Authentication

# Login to get JWT token
curl -X POST https://api.agl.com/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "email": "user@agl.com",
    "password": "your-password"
  }'

# Use the token in subsequent requests
curl https://api.agl.com/api/infrastructure/status \
  -H "Authorization: Bearer YOUR_TOKEN_HERE"

# Or use API Key
curl https://api.agl.com/api/infrastructure/status \
  -H "X-API-Key: YOUR_API_KEY_HERE"
// Using fetch API
const login = async () => {
  const response = await fetch('https://api.agl.com/api/auth/login', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      email: 'user@agl.com',
      password: 'your-password'
    })
  });

  const { access_token } = await response.json();

  // Store token
  localStorage.setItem('token', access_token);

  return access_token;
};

// Make authenticated request
const getData = async () => {
  const token = localStorage.getItem('token');

  const response = await fetch(
    'https://api.agl.com/api/infrastructure/status',
    {
      headers: {
        'Authorization': `Bearer ${token}`
      }
    }
  );

  return response.json();
};

Get Infrastructure Status

curl https://api.agl.com/api/infrastructure/status \
  -H "Authorization: Bearer YOUR_TOKEN"
const getInfrastructureStatus = async () => {
  const response = await fetch(
    'https://api.agl.com/api/infrastructure/status',
    {
      headers: {
        'Authorization': `Bearer ${localStorage.getItem('token')}`
      }
    }
  );

  if (!response.ok) {
    throw new Error(`HTTP error! status: ${response.status}`);
  }

  return response.json();
};

// Usage
const status = await getInfrastructureStatus();
console.log(status.summary);
import requests

def get_infrastructure_status(token):
    """Get infrastructure status"""
    url = 'https://api.agl.com/api/infrastructure/status'
    headers = {'Authorization': f'Bearer {token}'}

    response = requests.get(url, headers=headers)
    response.raise_for_status()

    return response.json()

# Usage
status = get_infrastructure_status(token)
print(status['summary'])
<?php
use GuzzleHttp\Client;

function getInfrastructureStatus(string $token): array
{
    $client = new Client([
        'base_uri' => 'https://api.agl.com/api',
    ]);

    $response = $client->get('/infrastructure/status', [
        'headers' => [
            'Authorization' => "Bearer $token"
        ]
    ]);

    return json_decode($response->getBody(), true);
}

// Usage
$status = getInfrastructureStatus($token);
print_r($status['summary']);
?>

const createContainer = async (config) => {
  const response = await fetch(
    'https://api.agl.com/api/containers/create',
    {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${localStorage.getItem('token')}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(config)
    }
  );

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
  }

  return response.json();
};

// Usage
const container = await createContainer({
  node: 'pve1',
  vmid: 105,
  hostname: 'app-container',
  cores: 2,
  memory: 2048,
  rootfs: 'local-zfs:32',
  net0: 'name=eth0,bridge=vmbr0,ip=dhcp',
  unprivileged: true,
  start: true
});
import requests

def create_container(token, config):
    """Create a new container"""
    url = 'https://api.agl.com/api/containers/create'
    headers = {
        'Authorization': f'Bearer {token}',
        'Content-Type': 'application/json'
    }

    response = requests.post(url, json=config, headers=headers)
    response.raise_for_status()

    return response.json()

# Usage
container = create_container(token, {
    'node': 'pve1',
    'vmid': 105,
    'hostname': 'app-container',
    'cores': 2,
    'memory': 2048,
    'rootfs': 'local-zfs:32',
    'net0': 'name=eth0,bridge=vmbr0,ip=dhcp',
    'unprivileged': True,
    'start': True
})
